First we create some high symmetry kpoints then we explicitely interpolate between the high symmetry kpoints to form bands_kpoints.

kpoints = [Vec3(0.0, 0.0, 0.5),
           Vec3(0.0, 0.5, 0.5),
           Vec3(0.5, 0.5, 0.5),
           Vec3(0.5, 0.5, 0.0),
           Vec3(0.5, 0.0, 0.0),
           Vec3(0.0, 0.0, 0.0)]
band_kpoints = eltype(kpoints)[]
for i = 1:length(kpoints)-1
    for Œ± in range(0, 1, 20)
        push!(band_kpoints, Vec3((1-Œ±) .* kpoints[i] .+ Œ± .* kpoints[i+1]))
    end
end
bands = wannierbands(hami, band_kpoints)

plot(bands)
